所在目录 /etc/samba

smb.conf.example 示例文件
vim /etc/samba/smb.conf
[aa] //共享ID
comment = Public Stuff //解释、说明
path = /aa //共享目录路径
public = yes //是否共享（是否能在网上邻居中查看到）

cat /etc/iscsi/initiatorname.iscsi //可查询到设备的iqn号
InitiatorName=iqn.1994-05.com.redhat:72f7d1628af9
//基于服务器的iqn号，创建iscsi设备
/iscsi create iqn.1994-05.com.redhat:72f7d1628af9
//基于块设备创建lun 1（对外表现）
/iscsi/iqn.1994-05.com.redhat:72f7d1628af9/tpg1/luns create /backstores/block/server1 1
//创建ACL，设置哪些客户端可以访问共享资源，并将lun映射给acl中配置的客户端
/iscsi/iqn.1994-05.com.redhat:72f7d1628af9/tpg1/acls create iqn.1991-05.com.microsoft:js324-025
//删除默认的socket
/iscsi/iqn.1994-05.com.redhat:72f7d1628af9/tpg1/portals delete 0.0.0.0 3260
//设置网络资源，服务器通过哪个IP地址和端口号共享资源
/iscsi/iqn.1994-05.com.redhat:72f7d1628af9/tpg1/portals create 192.168.1.100 3260
saveconfig //保存配置
exit //退出
//重启服务
Systemctl restart iscsi target
